Node root;

void histogramma(Node node, int level, int[] array){
    if (node == null)   // se sono alla fine di un ramo "il figlio mai creato" //1
        return;

    array[level] += node.key; //1

    histogramma(node.left, level+1, array) // Theta(j) j:nodi di sisistra
    histogramma(node.right, level+1, array) // Theta(k) k: nodi di destra
}
